import React, { useState } from "react";
import { applyDriver } from "../api";

function DriverApplication() {
  const [name,setName] = useState("");
  const [phone,setPhone] = useState("");
  const [vehicleType,setVehicleType] = useState("");

  const handleSubmit = async (e)=>{
    e.preventDefault();
    await applyDriver({name,phone,vehicleType});
    alert("Application Submitted!");
  }

  return (
    <div>
      <h2>Driver Application</h2>
      <form onSubmit={handleSubmit}>
        <input placeholder="Name" value={name} onChange={e=>setName(e.target.value)} required/>
        <input placeholder="Phone" value={phone} onChange={e=>setPhone(e.target.value)} required/>
        <input placeholder="Vehicle Type" value={vehicleType} onChange={e=>setVehicleType(e.target.value)} required/>
        <button type="submit">Submit</button>
      </form>
    </div>
  )
}

export default DriverApplication;
